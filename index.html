function renderTable(data) { const tbody = document.getElementById('corpo-tabela'); tbody.innerHTML = ''; data.forEach(item => { const row = document.createElement('tr'); row.classList.add(`nivel-${item.Nivel}`); // Criação das células para cada coluna // Exemplo: Tarefa, ID, etc. // Adicionar a linha à tabela tbody.appendChild(row); // Lógica para adicionar tarefas filhas (recursivamente) if (item.Subtarefas) { item.Subtarefas.forEach(subtarefa => { const subRow = document.createElement('tr'); subRow.classList.add(`nivel-${subtarefa.Nivel}`); // Criação das células para as subtarefas // Exemplo: Tarefa, ID, etc. tbody.appendChild(subRow); }); } }); }
